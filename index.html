// fallback (แสดงก่อนเสมอ) let data={ first:"309612", last2:"77", front3:["355","108"], back3:["868","424"] }; // render function render(date){ document.getElementById("date").innerText=date; document.getElementById("first").innerText=data.first; document.getElementById("front3").innerText=data.front3.join(","); document.getElementById("back3").innerText=data.back3.join(","); document.getElementById("last2").innerText=data.last2; } // 🔥 สำคัญ: แสดง fallback ทันที render("งวดล่าสุด"); // โหลด API async function loadData(){ try{ let res=await fetch("https://lotto.api.rayriffy.com/latest"); if(!res.ok) throw "error"; let json=await res.json(); if(!json || !json.response) throw "error"; let d=json.response; data.first=d.prizes.find(p=>p.id===1)?.number||data.first; data.last2=d.runningNumbers.find(x=>x.prize==="เลขท้าย 2 ตัว")?.number||data.last2; data.front3=d.runningNumbers.filter(x=>x.prize.includes("เลขหน้า")).map(x=>x.number)||data.front3; data.back3=d.runningNumbers.filter(x=>x.prize.includes("เลขท้าย 3 ตัว")).map(x=>x.number)||data.back3; // ✅ อัปเดตใหม่ render(d.date); }catch(e){ console.log("ใช้ fallback",e); } } loadData();
